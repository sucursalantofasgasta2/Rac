<script>
  document.getElementById("fecha").valueAsDate = new Date();

  function guardar() {
    const data = {
      responsable: responsable.value,
      fecha: fecha.value,
      pitbull: pitbull.checked,
      cerradas: cerradas.checked,
      bodegas: bodegas.checked,
      combustible: combustible.checked,
      basura: basura.checked,
      obs: obs.value
    };

    localStorage.setItem("cierre-" + data.fecha, JSON.stringify(data));
    generarPDF(data);
  }

  function generarPDF(d) {
    const { jsPDF } = window.jspdf;
    const pdf = new jsPDF();

    pdf.text("CIERRE DIARIO RENT A CAR", 20, 20);
    pdf.text("Fecha: " + d.fecha, 20, 30);
    pdf.text("Responsable: " + d.responsable, 20, 40);

    let y = 60;
    pdf.text("Checklist:", 20, y);
    y += 10;

    pdf.text(`Pitbull a las unidades: ${d.pitbull ? "OK" : "NO"}`, 20, y); y+=8;
    pdf.text(`Unidades cerradas: ${d.cerradas ? "OK" : "NO"}`, 20, y); y+=8;
    pdf.text(`Candados bodegas: ${d.bodegas ? "OK" : "NO"}`, 20, y); y+=8;
    pdf.text(`Estación combustible: ${d.combustible ? "OK" : "NO"}`, 20, y); y+=8;
    pdf.text(`Basura del lavado: ${d.basura ? "OK" : "NO"}`, 20, y); y+=10;

    pdf.text("Observaciones:", 20, y);
    pdf.text(d.obs || "-", 20, y + 8);

    pdf.save(`Cierre_${d.fecha}.pdf`);
  }
</script>
